## Init (prefix every script)

```typescript
const API_KEY = process.env.YOUTUBE_API_KEY!;
const BASE = 'https://www.googleapis.com/youtube/v3';

async function ytApi<T = any>(endpoint: string, params: Record<string, string | number>): Promise<T> {
  const qs = new URLSearchParams(Object.fromEntries(Object.entries(params).map(([k, v]) => [k, String(v)])));
  qs.set('key', API_KEY);
  const res = await fetch(`${BASE}/${endpoint}?${qs}`);
  if (!res.ok) {
    const body = await res.text();
    throw new Error(`YouTube API ${res.status}: ${body}`);
  }
  return res.json() as Promise<T>;
}
```

## Playlist Items — Enumerate Uploads (1 unit per 50 videos)

```typescript
const allVideos: any[] = [];
let pageToken = '';
const MAX_VIDEOS = 500;

while (allVideos.length < MAX_VIDEOS) {
  const params: Record<string, string | number> = { part: 'snippet,contentDetails', playlistId: 'UPLOADS_PLAYLIST_ID', maxResults: 50 };
  if (pageToken) params.pageToken = pageToken;
  const r = await ytApi('playlistItems', params);
  allVideos.push(...r.items.map((i: any) => ({
    videoId: i.contentDetails.videoId,
    title: i.snippet.title,
    publishedAt: i.contentDetails.videoPublishedAt || i.snippet.publishedAt
  })));
  pageToken = r.nextPageToken || '';
  if (!pageToken) break;
}
console.log(JSON.stringify({ total: allVideos.length, videos: allVideos }, null, 2));
```

## Art Track Parser

```typescript
function parseArtTrack(description: string): { distributor?: string; song?: string; artist?: string; album?: string; label?: string; releaseDate?: string } | null {
  if (!description.includes('Auto-generated by YouTube')) return null;
  const lines = description.split('\n').filter(l => l.trim());
  const result: any = {};

  // "Provided to YouTube by [Distributor]"
  const distMatch = lines[0]?.match(/Provided to YouTube by (.+)/);
  if (distMatch) result.distributor = distMatch[1].trim();

  // "Song · Artist" line
  const songLine = lines.find(l => l.includes(' · '));
  if (songLine) {
    const [song, ...artists] = songLine.split(' · ');
    result.song = song?.trim();
    result.artist = artists.join(', ').trim();
  }

  // Album — line after song line, before ℗
  const songIdx = lines.indexOf(songLine || '');
  if (songIdx >= 0 && songIdx + 1 < lines.length) {
    const nextLine = lines[songIdx + 1];
    if (nextLine && !nextLine.startsWith('℗') && !nextLine.startsWith('Released')) {
      result.album = nextLine.trim();
    }
  }

  // "℗ Year Label"
  const pLine = lines.find(l => l.startsWith('℗'));
  if (pLine) {
    const pMatch = pLine.match(/℗\s*(\d{4})\s+(.+)/);
    if (pMatch) result.label = pMatch[2].trim();
  }

  // "Released on: YYYY-MM-DD"
  const relLine = lines.find(l => l.startsWith('Released on:'));
  if (relLine) {
    const relMatch = relLine.match(/Released on:\s*(\d{4}-\d{2}-\d{2})/);
    if (relMatch) result.releaseDate = relMatch[1];
  }

  return Object.keys(result).length > 0 ? result : null;
}
```

## Duration Parser

```typescript
function parseDuration(iso: string): number {
  const m = iso.match(/PT(?:(\d+)H)?(?:(\d+)M)?(?:(\d+)S)?/);
  if (!m) return 0;
  return (Number(m[1] || 0) * 3600) + (Number(m[2] || 0) * 60) + Number(m[3] || 0);
}

function formatDuration(seconds: number): string {
  const h = Math.floor(seconds / 3600);
  const m = Math.floor((seconds % 3600) / 60);
  const s = seconds % 60;
  return h > 0 ? `${h}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}` : `${m}:${String(s).padStart(2,'0')}`;
}
```

## URL Parsers

```typescript
function parseYouTubeUrl(url: string): { type: string; id: string } | null {
  // Channel ID: youtube.com/channel/UCxxxxx
  let m = url.match(/youtube\.com\/channel\/(UC[a-zA-Z0-9_-]{22})/);
  if (m) return { type: 'channel', id: m[1] };

  // Handle: youtube.com/@handle
  m = url.match(/youtube\.com\/@([a-zA-Z0-9._-]+)/);
  if (m) return { type: 'handle', id: m[1] };

  // Video: youtube.com/watch?v=xxx or youtu.be/xxx
  m = url.match(/(?:youtube\.com\/watch\?v=|youtu\.be\/)([a-zA-Z0-9_-]{11})/);
  if (m) return { type: 'video', id: m[1] };

  return null;
}
```

## Full Pipeline (Parallel where possible)

Use this after obtaining `channelId` for a complete investigation:

```typescript
// Step 1: Channel profile
const chRes = await ytApi('channels', { part: 'snippet,statistics,contentDetails,topicDetails,brandingSettings', id: channelId });
const ch = chRes.items[0];
const uploadsId = ch.contentDetails.relatedPlaylists.uploads;

// Step 2: Enumerate uploads (paginated)
const allVideos: any[] = [];
let pageToken = '';
while (allVideos.length < 500) {
  const params: Record<string, string | number> = { part: 'snippet,contentDetails', playlistId: uploadsId, maxResults: 50 };
  if (pageToken) params.pageToken = pageToken;
  const r = await ytApi('playlistItems', params);
  allVideos.push(...r.items.map((i: any) => ({ videoId: i.contentDetails.videoId, title: i.snippet.title, publishedAt: i.contentDetails.videoPublishedAt })));
  pageToken = r.nextPageToken || '';
  if (!pageToken) break;
}

// Step 3: Batch video stats
const allStats: any[] = [];
const videoIds = allVideos.map(v => v.videoId);
for (let i = 0; i < videoIds.length; i += 50) {
  const batch = videoIds.slice(i, i + 50);
  const r = await ytApi('videos', { part: 'snippet,statistics,contentDetails,topicDetails', id: batch.join(',') });
  allStats.push(...r.items.map((v: any) => ({
    videoId: v.id, title: v.snippet.title, description: v.snippet.description,
    publishedAt: v.snippet.publishedAt, categoryId: v.snippet.categoryId,
    tags: v.snippet.tags || [],
    views: Number(v.statistics.viewCount || 0), likes: Number(v.statistics.likeCount || 0),
    comments: Number(v.statistics.commentCount || 0),
    duration: v.contentDetails.duration, licensedContent: v.contentDetails.licensedContent,
    topicCategories: v.topicDetails?.topicCategories || []
  })));
}

// Step 4: Compute metrics
const totalViews = allStats.reduce((s, v) => s + v.views, 0);
const totalLikes = allStats.reduce((s, v) => s + v.likes, 0);
const totalComments = allStats.reduce((s, v) => s + v.comments, 0);
const musicVideos = allStats.filter(v => v.categoryId === '10');
const licensedVideos = allStats.filter(v => v.licensedContent);
const topByViews = [...allStats].sort((a, b) => b.views - a.views).slice(0, 10);

console.log(JSON.stringify({
  channel: { id: ch.id, title: ch.snippet.title, customUrl: ch.snippet.customUrl,
    publishedAt: ch.snippet.publishedAt, country: ch.snippet.country,
    subscribers: Number(ch.statistics.subscriberCount), totalViews: Number(ch.statistics.viewCount),
    videoCount: Number(ch.statistics.videoCount),
    topicCategories: ch.topicDetails?.topicCategories || [],
    keywords: ch.brandingSettings?.channel?.keywords || '' },
  videos: { total: allStats.length, musicCategory: musicVideos.length,
    licensed: licensedVideos.length, totalViews, totalLikes, totalComments,
    avgViews: Math.round(totalViews / (allStats.length || 1)),
    engagementRate: totalViews > 0 ? ((totalLikes + totalComments) / totalViews * 100).toFixed(2) + '%' : '0%' },
  topVideos: topByViews
}, null, 2));
```
